procedure init()
{
	cset( 0, 255, 0, 0)
	cset( 1, 255,127,0)
	cset( 2, 255,255,0)
	cset( 3, 0,255,255)
	cset( 4, 0,255,0)
	cset( 5, 0,0,255)
	cset( 6, 75,0,130)
	cset( 7, 143,0,255)
}

procedure main()
{
var
	byte r, g, b
	byte i,j
	bool finish
evar
	init()
	draw
	finish = 1
	while(finish)
		r = CR0
		g = CG0
		b = CB0
		for(i, 0, 6)
			ccopy(i, i + 1)
		efor
		cset( 7, r, g, b)
		ccopyall(8)
		pause(100)
		draw
	ewhile
}
